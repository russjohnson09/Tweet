1   package model;
2   
3   import java.text.SimpleDateFormat;
4   import java.util.ArrayList;
5   import java.util.List;
6   
7   import javax.swing.AbstractListModel;
8   
9   import twitter4j.Status;
10  import twitter4j.Twitter;
11  import twitter4j.TwitterException;
12  
13  /****************************************************
14   * Tweets Class.
15   ***************************************************/
16  public class Tweets extends AbstractListModel<String> {
17  
18      /** twitter. */
19      private Twitter t;
20  
21      /** serialVersionUID. */
22      private static final long serialVersionUID = 1L;
23  
24      /** List tweets. */
25      private List<Status> tweets;
26  
27      /****************************************************
28       * Tweets Constructor.
29       * @param t Twitter
30       ***************************************************/
31      public Tweets(final Twitter t) {
32          super();
33          this.t = t;
34          tweets = new ArrayList<Status>();
35      }
36  
37      /****************************************************
38       * returns Tweets arraylist.
39       * @return tweets
40       ***************************************************/
41      public final List<Status> getTweets() {
42          return tweets;
43      }
44  
45      /****************************************************
46       * gets element at specified location.
47       * @param i int
48       * @return string
49       ***************************************************/
50      @Override
51      public final String getElementAt(final int i) {
52          Status s = tweets.get(i);
53  
54          SimpleDateFormat df = new SimpleDateFormat();
55          df.applyPattern("MM/dd/yyyy");
56  
57          String str = s.getUser().getName() + " (" + df.format(s.getCreatedAt())
58                  + ") --- " + s.getText();
59          return str;
60      }
61  
62      @Override
63      /****************************************************
64       * Gets size of tweet.
65       ***************************************************/
66      public final int getSize() {
67          return tweets.size();
68      }
69  
70      /****************************************************
71       * Adds tweet.
72       * @param s
73       *            Status
74       ***************************************************/
75      public final void add(final Status s) {
76          tweets.add(s);
77          fireIntervalAdded(this, 0, tweets.size());
78      }
79  
80      /****************************************************
81       * Removes Tweet.
82       * @param index int
83       * @return l long
84       ***************************************************/
85      public final long remove(final int index) {
86          long l = tweets.get(index).getId();
87          tweets.remove(index);
88          fireIntervalRemoved(this, 0, tweets.size());
89          return l;
90      }
91  
92      /****************************************************
93       * homeTimeline.
94       ***************************************************/
95      public final void homeTimeline() {
96          List<Status> statuses = null;
97          try {
98              statuses = t.getHomeTimeline();
99          } catch (TwitterException e) {
100             e.printStackTrace();
101         }
102         if (statuses != null) {
103             tweets.clear();
104             for (Status s : statuses) {
105                 tweets.add(s);
106                 fireContentsChanged(this, 0, tweets.size());
107             }
108         }
109     }
110 
111     /****************************************************
112      * User Timeline.
113      ***************************************************/
114     public final void userTimeline() {
115         List<Status> statuses = null;
116         try {
117             statuses = t.getUserTimeline();
118         } catch (TwitterException e) {
119             e.printStackTrace();
120         }
121         if (statuses != null) {
122             tweets.clear();
123             for (Status s : statuses) {
124                 tweets.add(s);
125                 fireContentsChanged(this, 0, tweets.size());
126             }
127         }
128     }
129 
130     /****************************************************
131      * Show Id.
132      * @param index int
133      * @return l long
134      ***************************************************/
135     public final long showId(final int index) {
136         return tweets.get(index).getId();
137     }
138 }
