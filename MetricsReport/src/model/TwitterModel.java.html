95      /****************************************************
96       * Refresh.
97       ***************************************************/
98      public final void refresh() {
99          List<Status> statuses;
100         followers = new ArrayList<User>();
101         following = new ArrayList<User>();
102         homeTimeline = new Tweets(t);
103         userTimeline = new Tweets(t);
104         try {
105             User u = t.showUser(t.getId());
106             name = u.getName();
107             screenName = "@" + u.getScreenName();
108             description = u.getDescription();
109             url = u.getURL();
110             location = u.getLocation();
111             tweetCount = u.getStatusesCount();
112             friendsCount = u.getFriendsCount();
113             followersCount = u.getFollowersCount();
114             trending = t.getPlaceTrends(1);
115 
116             statuses = t.getHomeTimeline();
117             if (statuses != null) {
118                 for (Status s : statuses) {
119                     homeTimeline.add(s);
120                 }
121             }
122             statuses = t.getUserTimeline();
123             if (statuses != null) {
124                 for (Status s : statuses) {
125                     userTimeline.add(s);
126                 }
127             }
128 
129             friendsIDs = t.getFriendsIDs(-1).getIDs();
130 
131             long[] list = t.getFollowersIDs(-1).getIDs();
132             for (long l : list) {
133                 followers.add(t.showUser(l));
134             }
135             list = t.getFriendsIDs(-1).getIDs();
136             for (long l : list) {
137                 following.add(t.showUser(l));
138             }
139             messages = t.getDirectMessages();
140 
141             refreshImage(u);
142 
143         } catch (IllegalStateException e) {
144             e.printStackTrace();
145         } catch (TwitterException e) {
146             e.printStackTrace();
147         }
148 
149     }
