323     /****************************************************
324      * Sets up controller.
325      ***************************************************/
326     private void setUpController() {
327         controller = new TwitterController();
328         if (!controller.getIsSetUp()) {
329             final String authUrl = controller.getAuthUrl();
330             final TextField pinTF = new TextField(10);
331             final JFrame authFrame = new JFrame();
332             JButton confirmPinButton = new JButton("OK");
333             confirmPinButton.setFocusable(false);
334             JButton browserButton = new JButton("Open Browser");
335             browserButton.setFocusable(false);
336 
337             /** Confirm Pin Button Action Listener */
338             confirmPinButton.addActionListener(new ActionListener() {
339                 @Override
340                 public void actionPerformed(final ActionEvent arg0) {
341                     if (pinTF.getText().length() > 0) {
342                         controller.setUp(pinTF.getText());
343                         if (controller.getIsSetUp()) {
344                             authFrame.setVisible(false);
345                             authFrame.dispose();
346                         }
347                     }
348                 }
349             });
350 
351             /** Browser Button Action Listener */
352             browserButton.addActionListener(new ActionListener() {
353                 @Override
354                 public void actionPerformed(final ActionEvent arg0) {
355                     if (Desktop.isDesktopSupported()) {
356                         authFrame.toFront();
357                         try {
358                             Desktop.getDesktop().browse(new URI(authUrl));
359                         } catch (IOException e) {
360                             e.printStackTrace();
361                         } catch (URISyntaxException e) {
362                             e.printStackTrace();
363                         }
364                     }
365                 }
366             });
367             JPanel pnl = new JPanel();
368 
369             final int AUTH_WIDTH = 500;
370             final int AUTH_HEIGHT = 200;
371 
372             authFrame.setSize(AUTH_WIDTH, AUTH_HEIGHT);
373             authFrame.setLocationRelativeTo(null);
374             pnl.add(new JLabel(
375                     "<html><center>Before you use this application, you "
376                             + "need to authorize your Twitter Account.<br>Click "
377                             + "below to open your browser and sign into your account."
378                             + "<br><br>Then copy the pin number that you receive and enter "
379                             + "it in the text box.<br>Then click \"OK\" and enjoy using "
380                             + " Desktop Tweets!<br><br><br></center></html>"));
381             pnl.add(browserButton);
382             pnl.add(new JLabel("     Enter pin:"));
383             pnl.add(pinTF);
384             pnl.add(confirmPinButton);
385             authFrame.add(pnl);
386             authFrame.setTitle("Authorization");
387             authFrame.setVisible(true);
388         }
389     }
