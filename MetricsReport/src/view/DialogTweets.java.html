3   import java.awt.BorderLayout;
4   import java.awt.Dimension;
5   import java.awt.event.ActionEvent;
6   import java.awt.event.ActionListener;
7   import java.util.ArrayList;
8   
9   import javax.swing.JButton;
10  import javax.swing.JDialog;
11  import javax.swing.JFrame;
12  import javax.swing.JList;
13  import javax.swing.JOptionPane;
14  import javax.swing.JScrollPane;
15  
16  import model.Tweets;
17  
18  /****************************************************
19   * DialogTweets Class.
20   ***************************************************/
21  public class DialogTweets extends JDialog implements ActionListener {
22  
23      /** Remove button. */
24      private JButton remove;
25  
26      /** Jlist of tweets. */
27      private JList<String> list;
28  
29      /** list of tweets to be removed. */
30      private ArrayList<Long> removeList = new ArrayList<Long>();
31  
32      /** Tweet. */
33      private Tweets tweets;
34  
35      /** Width of JDialog. */
36      private final int w = 400;
37      /** Length of JDialog. */
38      private final int l = 300;
39  
40      /****************************************************
41       * DialogTweets Class.
42       * 
43       * @param parent
44       *            JFrame
45       * @param t
46       *            Tweets
47       ***************************************************/
48      public DialogTweets(final JFrame parent, final Tweets t) {
49          super(parent, true);
50  
51          setLayout(new BorderLayout());
52  
53          tweets = t;
54  
55          setTitle("Tweets");
56          setPreferredSize(new Dimension(w, l));
57  
58          list = new JList<String>(tweets);
59  
60          JScrollPane scrollpane = new JScrollPane(list);
61          add(scrollpane, BorderLayout.CENTER);
62  
63          remove = new JButton("Remove");
64          remove.addActionListener(this);
65  
66          add(remove, BorderLayout.SOUTH);
67  
68          pack();
69          setVisible(true);
70      }
71  
72      @Override
73      public final void actionPerformed(final ActionEvent e) {
74          if (e.getSource() == remove) {
75              int i = list.getSelectedIndex();
76              if (i >= 0) {
77                  removeList.add(tweets.remove(i));
78              } else {
79                  JOptionPane.showMessageDialog(null,
80                          "Please first select a tweet", "Oops!",
81                          JOptionPane.PLAIN_MESSAGE);
82              }
83          }
84      }
85  
86      /****************************************************
87       * Removes list.
88       * 
89       * @return ArrayList<Lsong>
90       ***************************************************/
91      public final ArrayList<Long> getRemoveList() {
92          return removeList;
93      }
94  }
