73      /****************************************************
74       * Twitter Controller constructor.
75       ***************************************************/
76      public TwitterController() {
77          twitter = new TwitterFactory().getInstance();
78          twitter.setOAuthConsumer(CONSUMER_KEY, CONSUMER_KEY_SECRET);
79  
80          String token = getAccessToken();
81          String tokenSecret = getAccessTokenSecret();
82  
83          if (token != null && tokenSecret != null) {
84              twitter.setOAuthAccessToken(new AccessToken(token, tokenSecret));
85              isSetUp = true;
86  
87          } else {
88              try {
89                  requestToken = twitter.getOAuthRequestToken();
90              } catch (TwitterException e) {
91                  e.printStackTrace();
92              }
93          }
94  
95          model = new TwitterModel(twitter);
96      }

370     /****************************************************
371      * adds access token to file.
372      * @param accessToken AccessToken
373      ***************************************************/
374     @SuppressWarnings("unchecked")
375     private void addToFile(final AccessToken accessToken) {
376         JSONObject obj = new JSONObject();
377 
378         obj.put("AccessToken", accessToken.getToken());
379         obj.put("AccessTokenSecret", accessToken.getTokenSecret());
380 
381         try {
382             FileWriter file = new FileWriter("./auth.json");
383             file.write(obj.toJSONString());
384             file.flush();
385             file.close();
386 
387             isSetUp = true;
388 
389         } catch (IOException e) {
390             e.printStackTrace();
391         }
392 
393     }

395     /****************************************************
396      * Gets access token.
397      * @return String access token
398      ***************************************************/
399     private String getAccessToken() {
400         JSONParser parser = new JSONParser();
401         try {
402 
403             Object obj = parser.parse(new FileReader("./auth.json"));
404 
405             JSONObject jsonObject = (JSONObject) obj;
406 
407             return (String) jsonObject.get("AccessToken");
408 
409         } catch (Throwable e) {
410             return null;
411         }
412     }
